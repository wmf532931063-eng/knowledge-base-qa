### 1. 环境变量管理优化

#### 当前问题
- 环境变量分散在多个地方
- 线上环境配置不一致
- 缺少环境验证机制

#### 解决方案
创建统一的配置文件和环境验证脚本：

```javascript
// config/validateEnv.js
function validateEnvironment() {
  const requiredEnvVars = [
    'NEXT_PUBLIC_LLM_API_KEY',
    'NEXT_PUBLIC_LLM_BASE_URL',
    'PORT'
  ];
  
  const missingVars = requiredEnvVars.filter(varName => !process.env[varName]);
  
  if (missingVars.length > 0) {
    console.error('❌ 缺少必要环境变量:', missingVars.join(', '));
    return false;
  }
  
  // 验证API密钥格式
  const apiKey = process.env.NEXT_PUBLIC_LLM_API_KEY;
  if (apiKey.includes('your_') || apiKey.includes('xxx') || apiKey.length < 20) {
    console.error('❌ API密钥配置不正确');
    return false;
  }
  
  console.log('✅ 环境变量验证通过');
  return true;
}
```

```javascript
// server/services/enhancedLLMService.js
class EnhancedLLMService {
  constructor() {
    this.providers = [
      {
        name: '通义千问',
        call: this.callQwenAPI,
        priority: 1
      },
      {
        name: '腾讯云DeepSeek',
        call: this.callTencentAPI,
        priority: 2
      },
      {
        name: '本地模拟',
        call: this.callLocalSimulation,
        priority: 3
      }
    ];
  }
  
  async generateAnswer(question, context, maxRetries = 3) {
    // 按优先级尝试所有提供商
    for (const provider of this.providers.sort((a, b) => a.priority - b.priority)) {
      for (let attempt = 1; attempt <= maxRetries; attempt++) {
        try {
          console.log(`🔍 尝试使用 ${provider.name} (第${attempt}次尝试)`);
          const answer = await provider.call(question, context);
          console.log(`✅ ${provider.name} 调用成功`);
          return answer;
        } catch (error) {
          console.warn(`⚠️ ${provider.name} 调用失败:`, error.message);
          
          if (attempt < maxRetries) {
            // 指数退避重试
            const delay = Math.pow(2, attempt - 1) * 1000;
            await new Promise(resolve => setTimeout(resolve, delay));
          }
        }
      }
    }
    
    // 所有提供商都失败，返回友好的错误信息
    return `抱歉，目前无法获取大模型的回答。请稍后重试，或联系管理员。\n\n相关上下文：\n${context.substring(0, 200)}...`;
  }
  
  async callQwenAPI(question, context) {
    // 现有通义千问调用逻辑
    // 添加超时控制
    const controller = new AbortController();
    const timeoutId = setTimeout(() => controller.abort(), 10000); // 10秒超时
    
    try {
      const response = await fetch(process.env.NEXT_PUBLIC_LLM_BASE_URL + '/chat/completions', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${process.env.NEXT_PUBLIC_LLM_API_KEY}`
        },
        body: JSON.stringify({
          model: process.env.NEXT_PUBLIC_LLM_MODEL || 'qwen-turbo',
          messages: [
            {
              role: 'system',
              content: `你是命运主理人。请根据以下参考信息回答：\n\n${context}\n\n回答要专业、准确，不要使用占卜格式。`
            },
            { role: 'user', content: question }
          ],
          max_tokens: 1000
        }),
        signal: controller.signal
      });
      
      clearTimeout(timeoutId);
      
      if (!response.ok) {
        throw new Error(`API响应状态: ${response.status}`);
      }
      
      const data = await response.json();
      return data.choices?.[0]?.message?.content || '未获得有效回答';
    } catch (error) {
      clearTimeout(timeoutId);
      throw error;
    }
  }
  
  async callTencentAPI(question, context) {
    // 腾讯云API调用逻辑
    // 作为备选方案
    // ...
  }
  
  async callLocalSimulation(question, context) {
    // 本地模拟回答
    const summary = context.split('\n\n').slice(0, 2).join('\n\n');
    return `基于本地知识库的简化回答：\n\n${summary}\n\n（大模型服务暂时不可用）`;
  }
  
  async testConnection() {
    // 测试所有提供商连接
    const results = [];
    
    for (const provider of this.providers) {
      try {
        const startTime = Date.now();
        await provider.call('测试连接', '这是一个连接测试');
        const endTime = Date.now();
        
        results.push({

          status: 'connected',
          latency: endTime - startTime
        });
      } catch (error) {
        results.push({
          provider: provider.name,
          status: 'failed',
          error: error.message
        });
      }
    }
    
    return results;
  }
```

### 4. 自动监控和恢复机制

#### 解决方案
创建监控脚本，定期检查系统状态：

```javascript
// scripts/monitor.js
const cron = require('node-cron');
const EnhancedLLMService = require('../server/services/enhancedLLMService');
const fs = require('fs');
const path = require('path');

class SystemMonitor {
  constructor() {
    this.llmService = new EnhancedLLMService();
    this.statusFile = path.join(__dirname, '../logs/system-status.json');
    this.checkInterval = '*/5 * * * *'; // 每5分钟检查一次
  }
  
  async checkSystemHealth() {
    const healthReport = {
      timestamp: new Date().toISOString(),
      checks: []
    };
    
    // 1. 检查大模型连接
    try {
      const llmResults = await this.llmService.testConnection();
      healthReport.checks.push({
        name: '大模型连接',
        status: llmResults.some(r => r.status === 'connected') ? 'healthy' : 'degraded',
        details: llmResults
      });
    } catch (error) {
      healthReport.checks.push({
        name: '大模型连接',
        status: 'failed',
        error: error.message
      });
    }
    
    // 2. 检查知识库状态
    try {
      const statsPath = path.join(__dirname, '../vector-store/stats.json');
      if (fs.existsSync(statsPath)) {
        const stats = JSON.parse(fs.readFileSync(statsPath, 'utf8'));
        healthReport.checks.push({
          name: '知识库',
          status: stats.documentCount > 0 ? 'healthy' : 'empty',
          details: stats
        });
      } else {
        healthReport.checks.push({
          name: '知识库',
          status: 'missing',
          error: '知识库文件不存在'
        });
      }
    } catch (error) {
      healthReport.checks.push({
        name: '知识库',
        status: 'failed',
        error: error.message
      });
    }
    
    // 3. 检查服务器资源
    try {
      const memoryUsage = process.memoryUsage();
      healthReport.checks.push({
        name: '服务器资源',
        status: memoryUsage.heapUsed / memoryUsage.heapTotal < 0.8 ? 'healthy' : 'warning',
        details: {
          memoryUsed: Math.round(memoryUsage.heapUsed / 1024 / 1024) + 'MB',
          memoryTotal: Math.round(memoryUsage.heapTotal / 1024 / 1024) + 'MB',
          uptime: process.uptime()
        }
      });
    } catch (error) {
      healthReport.checks.push({
        name: '服务器资源',
        status: 'failed',
        error: error.message
      });
    }
    
    // 保存健康报告
    this.saveHealthReport(healthReport);
    
    // 自动恢复逻辑
    await this.autoRecovery(healthReport);
    
    return healthReport;
  }
  
  saveHealthReport(report) {
    try {
      const logsDir = path.dirname(this.statusFile);
      if (!fs.existsSync(logsDir)) {
        fs.mkdirSync(logsDir, { recursive: true });
      }
      
      // 读取历史记录
      let history = [];
      if (fs.existsSync(this.statusFile)) {
        history = JSON.parse(fs.readFileSync(this.statusFile, 'utf8'));
      }
      
      // 保留最近100条记录
      history.push(report);
      if (history.length > 100) {
        history = history.slice(-100);
      }
      
      fs.writeFileSync(this.statusFile, JSON.stringify(history, null, 2));
    } catch (error) {
      console.error('保存健康报告失败:', error.message);
    }
  }
  
  async autoRecovery(healthReport) {
    // 根据健康报告自动恢复
    const failedChecks = healthReport.checks.filter(check => check.status === 'failed');
    
    for (const check of failedChecks) {
      switch (check.name) {
        case '知识库':
          console.log('🔄 检测到知识库问题，尝试自动重建...');
          // 调用知识库重建逻辑
          break;
        case '大模型连接':
          console.log('🔄 检测到大模型连接问题，尝试重新连接...');
          // 尝试重新连接或切换提供商
          break;
      }
    }
  }
  
  start() {
    console.log('🚀 启动系统监控...');
    
    // 定时检查
    cron.schedule(this.checkInterval, async () => {
      console.log('🔍 执行定期系统检查...');
      const report = await this.checkSystemHealth();
      
      // 如果有严重问题，发送通知
      const criticalIssues = report.checks.filter(check => 
        check.status === 'failed' || check.name === '大模型连接' && check.status !== 'healthy'
      );
      
      if (criticalIssues.length > 0) {
        console.warn('⚠️ 检测到系统问题:', criticalIssues.map(issue => issue.name).join(', '));
        // 这里可以添加通知逻辑，如发送邮件、Slack消息等
      }
    });
    
    // 立即执行一次检查
    this.checkSystemHealth();
  }
}
```

### 6. 实现备份和快速恢复

#### 创建备份脚本
`scripts/backup.js`:

```javascript
const fs = require('fs');
const path = require('path');
const { execSync } = require('child_process');

class BackupManager {
  constructor() {
    this.backupDir = path.join(__dirname, '../backups');
    this.maxBackups = 5;
  }
  
  async createBackup() {
    const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
    const backupPath = path.join(this.backupDir, `backup-${timestamp}`);
    
    try {
      // 创建备份目录
      fs.mkdirSync(backupPath, { recursive: true });
      
      // 备份关键文件
      const filesToBackup = [
        'vector-store',
        'knowledge-base',
        '.env',
        'server/services',
        'server/routes'
      ];
      
      for (const file of filesToBackup) {
        const source = path.join(__dirname, '../', file);
        if (fs.existsSync(source)) {
          const dest = path.join(backupPath, file);
          if (fs.lstatSync(source).isDirectory()) {
            this.copyDir(source, dest);
          } else {
            fs.copyFileSync(source, dest);
          }
        }
      }
      
      console.log(`✅ 备份创建成功: ${backupPath}`);
      
      // 清理旧备份
      this.cleanupOldBackups();
      
      return backupPath;
    } catch (error) {
      console.error('❌ 备份创建失败:', error.message);
      throw error;
    }
  }
  
  async restoreFromBackup(backupPath) {
    if (!fs.existsSync(backupPath)) {
      throw new Error('备份文件不存在');
    }
    
    try {
      console.log(`🔄 从备份恢复: ${backupPath}`);
      
      // 停止当前服务（如果运行中）
      try {
        execSync('pkill -f "node.*server"', { stdio: 'ignore' });
      } catch (error) {
        // 忽略错误，可能服务未运行
      }
      
      // 恢复文件
      const filesToRestore = fs.readdirSync(backupPath);
      for (const file of filesToRestore) {
        const source = path.join(backupPath, file);
        const dest = path.join(__dirname, '../', file);
        
        if (fs.existsSync(dest)) {
          fs.rmSync(dest, { recursive: true, force: true });
        }
        
        if (fs.lstatSync(source).isDirectory()) {
          this.copyDir(source, dest);
        } else {
          fs.copyFileSync(source, dest);
        }
      }
      
      console.log('✅ 备份恢复完成');
      return true;
    } catch (error) {
      console.error('❌ 备份恢复失败:', error.message);
      throw error;
    }
  }
  
  copyDir(src, dest) {
    fs.mkdirSync(dest, { recursive: true });
    const entries = fs.readdirSync(src, { withFileTypes: true });
    
    for (const entry of entries) {
      const srcPath = path.join(src, entry.name);
      const destPath = path.join(dest, entry.name);
      
      if (entry.isDirectory()) {
        this.copyDir(srcPath, destPath);
      } else {
        fs.copyFileSync(srcPath, destPath);
      }
    }
  }
  
  cleanupOldBackups() {
    if (!fs.existsSync(this.backupDir)) return;
    
    const backups = fs.readdirSync(this.backupDir)
      .filter(name => name.startsWith('backup-'))
      .map(name => ({
        name,
        path: path.join(this.backupDir, name),
        time: fs.statSync(path.join(this.backupDir, name)).mtime.getTime()
      }))
      .sort((a, b) => b.time - a.time);
    
    if (backups.length > this.maxBackups) {
      const toDelete = backups.slice(this.maxBackups);
      for (const backup of toDelete) {
        fs.rmSync(backup.path, { recursive: true, force: true });
        console.log(`🗑️ 删除旧备份: ${backup.name}`);
      }
    }
  }
}
```
